                var commentSchema = new Schema({
                    rating: {
                        type: Number,
                        min: 1,
                        max: 5,
                        required: true
                    },
                    comment: {
                        type: String,
                        required: true
                    },
                    author: {
                        type: String,
                        required: true
                    },
                }, {
                        timestamps: true
                    });
    
                const productSchema = new Schema({
                    name: {
                        type: String,
                        required: true,
                        unique: true
                    },
                    description: {
                        type: String,
                        required: true
                    },
                    image: {
                        type: String,
                        required: true
                    },
                    category: {
                        type: String,
                        required: true
                    },
                    label: {
                        type: String,
                        default: ''
                    },
                    featured: {
                        type: Boolean,
                        default: false
                    },
                    comments: [commentSchema]
                }, {
                        timestamps: true
                    });


                var Products = mongoose.model('product', productSchema);
